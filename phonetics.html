function analyzeCluster() { const cluster = document .getElementById("clusterInput") .value .trim() .split(""); const type = document.getElementById("clusterType").value; // Clear all boxes const ids = [ "preInitial","initial","postInitial", "preFinal","final","postFinal1","postFinal2","postFinal3" ]; ids.forEach(id => document.getElementById(id).value = ""); // ---------------- INITIAL CLUSTERS ---------------- if (type === "initial") { if (cluster.length === 1) { document.getElementById("initial").value = cluster[0]; } if (cluster.length === 2) { if (cluster[0] === "s") { document.getElementById("preInitial").value = "s"; document.getElementById("initial").value = cluster[1]; } else { document.getElementById("initial").value = cluster[0]; document.getElementById("postInitial").value = cluster[1]; } } if (cluster.length === 3 && cluster[0] === "s") { document.getElementById("preInitial").value = cluster[0]; document.getElementById("initial").value = cluster[1]; document.getElementById("postInitial").value = cluster[2]; } } // ---------------- FINAL CLUSTERS ---------------- if (type === "final") { const preFinalSet = ["m", "n", "ŋ", "l", "s"]; let remaining = [...cluster]; if (remaining.length && preFinalSet.includes(remaining[0])) { document.getElementById("preFinal").value = remaining.shift(); } if (remaining.length) document.getElementById("final").value = remaining.shift(); if (remaining.length) document.getElementById("postFinal1").value = remaining.shift(); if (remaining.length) document.getElementById("postFinal2").value = remaining.shift(); if (remaining.length) document.getElementById("postFinal3").value = remaining.shift(); } }
